#pragma once

#include <chrono>
#include "inputs.h"
#include "outputs.h"

//------------------------------------------------------------
// TRAITS DECLARATION
//------------------------------------------------------------

template<typename T>
struct Translator;


//------------------------------------------------------------
// OUTPUT TYPE ALIAS (THE "UPGRADE")
//------------------------------------------------------------

template<typename T>
using translated_t = typename Translator<T>::Output;


//------------------------------------------------------------
// TRANSLATE FUNCTION
//------------------------------------------------------------

template<typename T>
translated_t<T> translate(const T& input)
{
    return Translator<T>::translate(input);
}



//------------------------------------------------------------
// Sensor
//------------------------------------------------------------

template<>
struct Translator<SensorInput>
{
    using Output = SensorRecord;

    static Output translate(const SensorInput& in)
    {
        return {
            in.id,
            in.temperature,
            in.humidity,
            std::chrono::system_clock::now(),
            true
        };
    }
};

//------------------------------------------------------------
// Packet
//------------------------------------------------------------

template<>
struct Translator<NetworkPacket>
{
    using Output = PacketLog;

    static Output translate(const NetworkPacket& in)
    {
        return {
            in.source,
            in.destination,
            in.payload,
            in.payload.size(),
            true
        };
    }
};

//------------------------------------------------------------
// Command
//------------------------------------------------------------

template<>
struct Translator<UserCommand>
{
    using Output = CommandTask;

    static Output translate(const UserCommand& in)
    {
        static int nextId = 1;

        return {
            in.user,
            in.command,
            in.priority,
            nextId++,
            true
        };
    }
};


//-----------------------------------
// main.cpp


#include <iostream>
#include "translator_traits.h"

int main()
{
    SensorInput sensor{1, 23.4, 40.1};
    NetworkPacket packet{10, 20, "HELLO"};
    UserCommand cmd{"alice", "restart", 5};

    auto sensorOut = translate(sensor);
    auto packetOut = translate(packet);
    auto cmdOut = translate(cmd);

    std::cout << sensorOut.id << "\n";
    std::cout << packetOut.payloadSize << "\n";
    std::cout << cmdOut.taskId << "\n";
}


Why the Upgrade (translated_t) Matters
Without it:
-----------
auto out = translate(input);
-----------

But in templates you often need:
-----------
typename Translator<T>::Output
-----------

The alias simplifies that to:
-----------
translated_t<T>
-----------

Example:
-----------
template<typename T>
translated_t<T> process(const T& input)
{
    return translate(input);
}
-----------
This becomes very useful in large template systems.

#include <iostream>
#include <string>
#include <tuple>
#include <chrono>

//////////////////////////////////////////////////////////////
// outputs.h (simulated)
//////////////////////////////////////////////////////////////

struct SensorRecord
{
    int id;
    double temperature;
    double humidity;
    std::chrono::system_clock::time_point timestamp;
    bool valid;

    static SensorRecord generateDummy()
    {
        std::cout << "Generating SensorRecord\n";
        return {1, 22.5, 55.0, std::chrono::system_clock::now(), true};
    }
};

struct PacketLog
{
    int source;
    int destination;
    std::string payload;
    size_t payloadSize;
    bool checksumOk;

    static PacketLog generateDummy()
    {
        std::cout << "Generating PacketLog\n";
        return {10, 20, "HELLO", 5, true};
    }
};

struct CommandTask
{
    std::string user;
    std::string command;
    int priority;
    int taskId;
    bool queued;

    static CommandTask generateDummy()
    {
        std::cout << "Generating CommandTask\n";
        return {"alice", "restart_service", 5, 1001, true};
    }
};

//////////////////////////////////////////////////////////////
// Compile-time "loop over types"
//////////////////////////////////////////////////////////////

template<typename... Ts>
auto generateAllDummy()
{
    return std::tuple<Ts...>{ Ts::generateDummy()... };
}

//////////////////////////////////////////////////////////////
// Helper to iterate tuple at runtime
//////////////////////////////////////////////////////////////

template<typename Tuple, typename Func, std::size_t... I>
void tupleForEachImpl(Tuple& t, Func f, std::index_sequence<I...>)
{
    (f(std::get<I>(t)), ...);
}

template<typename Tuple, typename Func>
void tupleForEach(Tuple& t, Func f)
{
    tupleForEachImpl(
        t,
        f,
        std::make_index_sequence<std::tuple_size_v<Tuple>>{}
    );
}

//////////////////////////////////////////////////////////////
// main.cpp
//////////////////////////////////////////////////////////////

int main()
{
    // Compile-time "loop"
    auto messages = generateAllDummy<
        SensorRecord,
        PacketLog,
        CommandTask
    >();

    std::cout << "\nNow iterating the generated objects:\n";

    // Runtime iteration over generated objects
    tupleForEach(messages, [](auto& msg)
    {
        std::cout << "Processing message object\n";
    });

    return 0;
}

//--------------------------
--------
generateAllDummy<SensorRecord, PacketLog, CommandTask>()
--------

is expanded by the compiler into something equivalent to:
--------
std::tuple<SensorRecord, PacketLog, CommandTask>{
    SensorRecord::generateDummy(),
    PacketLog::generateDummy(),
    CommandTask::generateDummy()
};
--------

So the compiler effectively generates code like:
--------
SensorRecord::generateDummy()
PacketLog::generateDummy()
CommandTask::generateDummy()
--------

//------------------------------------------------------
//------------------------------------------------------
//------------------------------------------------------
// Simpler
//------------------------------------------------------
//------------------------------------------------------
//------------------------------------------------------
//------------------------------------------------------
If you only want to call the functions, not store the results:
--------
template<typename... Ts>
void generateAllDummy()
{
    (Ts::generateDummy(), ...);
}
--------

Usage:
--------
generateAllDummy<
    SensorRecord,
    PacketLog,
    CommandTask
>();
--------
